99      /**
100      * Fluent API that pre-configures HttpSecurity with SAML specific configuration.
101      *
102      * @param http HttpSecurity instance
103      * @return Same HttpSecurity instance
104      * @throws Exception
105      */
106     // CSRF must be disabled when processing /saml/** to prevent "Expected CSRF token not found" exception.
107     // See: http://stackoverflow.com/questions/26508835/spring-saml-extension-and-spring-security-csrf-protection-conflict/26560447
108     protected final HttpSecurity samlizedConfig(final HttpSecurity http) throws Exception {
109         http.httpBasic().authenticationEntryPoint(samlEntryPoint())
110                 .and()
111                 .csrf().ignoringAntMatchers("/saml/**")
112                 .and()
113                 .authorizeRequests().antMatchers("/saml/**").permitAll()
114                 .and()
115                 .addFilterBefore(metadataGeneratorFilter(), ChannelProcessingFilter.class)
116                 .addFilterAfter(filterChainProxy(), BasicAuthenticationFilter.class);
117 
118         return http;
119     }

139     // Entry point to initialize authentication
140     @Bean
141     public SAMLEntryPoint samlEntryPoint() {
142         WebSSOProfileOptions webSSOProfileOptions = new WebSSOProfileOptions();
143 
144         // Disable element scoping when sending requests to IdP to prevent
145         // "Response has invalid status code urn:oasis:names:tc:SAML:2.0:status:Responder, status message is null"
146         // exception
147         webSSOProfileOptions.setIncludeScoping(false);
148 
149         // Always use HTTP-Redirect instead of HTTP-Post, although both works with ADFS
150         webSSOProfileOptions.setBinding(SAMLConstants.SAML2_REDIRECT_BINDING_URI);
151 
152         // Force IdP to re-authenticate user if issued token is too old to prevent
153         // "Authentication statement is too old to be used with value" exception
154         // See: http://stackoverflow.com/questions/30528636/saml-login-errors
155         webSSOProfileOptions.setForceAuthN(true);
156 
157         // Determine what authentication method to use (WIA, user/password, etc).
158         // If not set, it will use authentication method order defined by IdP
159         if (!samlConfigBean().getAuthnContexts().isEmpty()) {
160             webSSOProfileOptions.setAuthnContexts(samlConfigBean().getAuthnContexts());
161         }
162 
163         SAMLEntryPoint samlEntryPoint = new SAMLEntryPoint();
164         samlEntryPoint.setDefaultProfileOptions(webSSOProfileOptions);
165 
166         return samlEntryPoint;
167     }

169     // Filter automatically generates default SP metadata
170     @Bean
171     public MetadataGeneratorFilter metadataGeneratorFilter() {
172         // generates base URL that matches `SAMLContextProviderLB` configuration
173         // to ensure SAML endpoints work for server doing SSL termination
174         StringBuilder sb = new StringBuilder();
175         sb.append("https://").append(samlConfigBean().getSpServerName());
176         if (samlConfigBean().getSpHttpsPort() != 443) {
177             sb.append(":").append(samlConfigBean().getSpHttpsPort());
178         }
179         sb.append(samlConfigBean().getSpContextPath());
180         String entityBaseUrl = sb.toString();
181 
182         MetadataGenerator metadataGenerator = new MetadataGenerator();
183         metadataGenerator.setKeyManager(keyManager());
184         metadataGenerator.setEntityBaseURL(entityBaseUrl);
185         return new MetadataGeneratorFilter(metadataGenerator);
186     }

309     // IDP Metadata configuration - paths to metadata of IDPs in circle of trust is here
310     @Bean
311     public CachingMetadataManager metadata() throws MetadataProviderException {
312         HTTPMetadataProvider httpMetadataProvider = new HTTPMetadataProvider(new Timer(true),
313                                                                              httpClient(),
314                                                                              getMetdataUrl());
315         httpMetadataProvider.setParserPool(parserPool());
316 
317         ExtendedMetadataDelegate extendedMetadataDelegate = new ExtendedMetadataDelegate(httpMetadataProvider);
318         // Disable metadata trust check to prevent "Signature trust establishment failed for metadata entry" exception
319         extendedMetadataDelegate.setMetadataTrustCheck(false);
320 
321         return new CachingMetadataManager(ImmutableList.<MetadataProvider>of(extendedMetadataDelegate));
322     }

334     // In order to get SAML to work for Sp servers doing SSL termination, `SAMLContextProviderLB` has
335     // to be used instead of `SAMLContextProviderImpl` to prevent the following exception:-
336     //
337     // "SAML message intended destination endpoint 'https://server/app/saml/SSO' did not match the
338     // recipient endpoint 'http://server/app/saml/SSO'"
339     //
340     // This configuration will work for Sp servers (not) doing SSL termination.
341     @Bean
342     public SAMLContextProviderLB contextProvider() {
343         SAMLContextProviderLB contextProviderLB = new SAMLContextProviderLB();
344         contextProviderLB.setScheme("https");
345         contextProviderLB.setServerName(samlConfigBean().getSpServerName());
346         contextProviderLB.setServerPort(samlConfigBean().getSpHttpsPort());
347         contextProviderLB.setIncludeServerPortInRequestURL(samlConfigBean().getSpHttpsPort() != 443);
348         contextProviderLB.setContextPath(samlConfigBean().getSpContextPath());
349         return contextProviderLB;
350     }
